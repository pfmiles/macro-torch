function macroTorch.classMetatable(cls, fieldMapName)
    return {
        __index = function(t, k)
            if fieldMapName and macroTorch[fieldMapName] and macroTorch[fieldMapName][k] ~= nil then
                return macroTorch[fieldMapName][k](t)
            end
            return cls[k]
        end
    }
end
